**Vanilla `<script>`:**
```js
// Mobile menu toggle
const hamburger = document.getElementById('nav-hamburger');
const menu = document.getElementById('mobile-menu');
const close = document.getElementById('mobile-close');

hamburger?.addEventListener('click', () => {
  menu?.classList.remove('hidden');
  document.body.classList.add('nav-open');
});

close?.addEventListener('click', () => {
  menu?.classList.add('hidden');
  document.body.classList.remove('nav-open');
});

// Close on link click
menu?.querySelectorAll('a').forEach(link => {
  link.addEventListener('click', () => {
    menu?.classList.add('hidden');
    document.body.classList.remove('nav-open');
  });
});

// Services dropdown on mobile (tap to toggle)
const servicesToggle = document.getElementById('services-toggle-mobile');
const servicesSubmenu = document.getElementById('services-submenu-mobile');
servicesToggle?.addEventListener('click', (e) => {
  e.preventDefault();
  servicesSubmenu?.classList.toggle('hidden');
});
```
