node -e "
const { chromium } = require('playwright');
(async () => {
  const browser = await chromium.launch();
  const pages = [
    { slug: 'coaching', path: '/services/ai-coaching-for-leaders' },
    { slug: 'training', path: '/services/ai-training-for-teams' },
    { slug: 'fractional', path: '/services/fractional-exec-support' },
  ];
  const viewports = [
    { name: 'desktop', width: 1920, height: 1080 },
    { name: 'tablet', width: 768, height: 1024 },
    { name: 'mobile', width: 390, height: 844 },
  ];
  for (const pg of pages) {
    for (const vp of viewports) {
      const page = await browser.newPage({ viewport: { width: vp.width, height: vp.height } });
      await page.goto('http://localhost:4321' + pg.path, { waitUntil: 'networkidle' });
      await page.waitForTimeout(1000);
      await page.screenshot({ path: pg.slug + '-' + vp.name + '.png', fullPage: true });
      await page.close();
    }
  }
  await browser.close();
})();
"
